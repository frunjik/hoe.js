  6 /**
  7  * @namespace creates new DOM elements
  8  * @param {String} tag name of tag to be created
  9  * @param [param[]] can take any number of params, {@link hoe.jquery_plugin}
 10  * @returns {jQuery}
 11  */
 12 var hoe = function(tag){
 13     var $ele = jQuery(document.createElement(tag));
 14     return $ele.hoe.apply($ele, Array.prototype.slice.call(arguments, 1));
 15 };

 17 /**
 18  * Exposed to any jQuery object as 'hoe' plugin.
 19  * Guess which jQuery method should be applied to object,
 20  * operation depends on param type.
 21  * @param {String} [param] append as text to element content
 22  * @param {Object} [param] set as html element attributes
 23  * @param {DOMElement[]|jQuery[]} [param] append param into element content
 24  * @returns {jQuery}
 25  */
 26 hoe.jquery_plugin = function() {
 27     /*
 28        to arguments based on type:
 29        - string: append text
 30        - plain object: set html attributes
 31        - DOM element & jQuery object: append elements
 32        - array: append elements
 33     */
 34 
 35     function _guess_apply($ele, param){
 36         var type = jQuery.type(param);
 37         if(type === "string"){
 38             $ele.append(param);
 39         }
 40         else if(type === "array"){
 41             for(var i=0, max=param.length; i<max; i++) {
 42                 $ele.append(param[i]);
 43             }
 44         }
 45         else if (type === "object") {
 46             if (jQuery.isPlainObject(param)){
 47                 for(var name in param){
 48                     $ele.attr(name, param[name]);
 49                 }
 50             }
 51             else if((param instanceof jQuery) || (param instanceof Element)){
 52                 $ele.append(param);
 53             }
 54             else {
 55                 throw Error("Invalid object: " + param);
 56             }
 57         }
 58         else {
 59             throw Error("Invalid type: " + type);
 60         }
 61     }
 62 
 63     for(var i=0, max=arguments.length; i<max; i++) {
 64         _guess_apply(this, arguments[i]);
 65     }
 66     return this;
 67 };
 68 
 69 jQuery.fn.hoe = hoe.jquery_plugin;

 85 /**
 86  * define factory functions to create DOM elements
 87  * @param {Object} [namespace=window] namespace where functions will be created
 88  * @param {String[]} [tags=hoe.init_default_tags] functions will be created
 89           for given tags
 90  */
 91 hoe.init = function(namespace, tags){
 92     namespace = namespace || window;
 93     tags = tags || hoe.init_default_tags;
 94 
 95     function _create_function(tag){
 96         return function(){
 97             var $ele = jQuery(document.createElement(tag));
 98             return $ele.hoe.apply($ele, arguments);
 99         };
100     }
101     for (var i=0, max=tags.length; i<max; i++){
102         namespace[tags[i]] = _create_function(tags[i]);
103     }
104 };

107 /**
108  * create new type by inheriting from other types
109  * @param {Object} base_type Type to inherit from
110  * @param {Function} [constructor] over-write constructor from base_type
111  * @return {Type} new Type/constructor
112  */
113 hoe.inherit = function (base_type, constructor){
114     var new_type;
115     if (constructor){
116         new_type = constructor;
117     }
118     else{
119         new_type = function(){return base_type.apply(this, arguments);};
120     }
121     $.extend(new_type, base_type);
122     $.extend(new_type.prototype, base_type.prototype);
123     return new_type;
124 };
125 
126 /**
127  * Base Type with helpers that bind scope to object.
128  * Constructor can be used to create new types.
129  * @constructor
130  * @param {Function} constructor constructor of a new Type
131  * @returns {Function/Type} new Type that inherit from hoe.Type
132  */
133 hoe.Type = function(constructor){
134     return hoe.inherit(hoe.Type, constructor);
135 };

138 /**
139  * Attach a callback on this object scope for event from observed
140  * @param {jQuery|hoe.Type} observed object that trigger events
141  * @param {String} event name of the event
142  * @param {Function} callback to be executed when event is triggered.
143  *                      this will be bound to current object.
144  */
145 hoe.Type.prototype.on = function(observed, event, callback){
146     if(observed instanceof jQuery){
147         observed.bind(event, $.proxy(callback, this));
148     }
149     else{
150         if (typeof observed._hoe_obs === 'undefined'){
151             observed._hoe_obs = {};
152         }
153         if (!(event in observed._hoe_obs)){
154             observed._hoe_obs[event] = [];
155         }
156         observed._hoe_obs[event].push({scope:this, fn:callback});
157     }
158 };
159 
160 /**
161  * Trigger an event for this object
162  * @param {String} event name of the event
163  * @param [arguments] other arguments will be passed to the callback
164  */
165 hoe.Type.prototype.trigger = function(event){
166     if (this._hoe_obs && this._hoe_obs[event]){
167         var callbacks = this._hoe_obs[event];
168         for (var i=0, max=callbacks.length; i<max; i++){
169             callbacks[i].fn.apply(callbacks[i].scope,
170                                   Array.prototype.slice.call(arguments, 1));
171         }
172     }
173 };

175 /**
176  * Iterate through Arrays or plain objects keeping the scope
177  * @param {Array|Object} seq contains items that will be iterated
178  * @param {Function} callback to be called for each item.
179           The callback the 3 paramaters:
180           1) value
181           2) index/key
182           3) reference to the whole sequence
183  */
184 hoe.Type.prototype.forEach = function(seq, fn){
185     if (jQuery.type(seq) == 'array'){
186         for(var i = 0, len = seq.length; i < len; ++i) {
187             fn.call(this, seq[i], i, seq);
188         }
189     }
190     else { // must be an object
191         for (var key in seq){
192             fn.call(this, seq[key], key, seq);
193         }
194     }
195 };
